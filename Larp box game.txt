  <button onclick="move(0,-1)">⬆</button>
  <button onclick="move(-1,0)">⬅</button>
  <button onclick="move(1,0)">➡</button>
  <button onclick="move(0,1)">⬇</button>
  <button onclick="attack()">⚔ Attack</button>
</div>

<script>
const canvas = document.getElementById("game");
const ctx = canvas.getContext("2d");
const mini = document.getElementById("minimap");
const mctx = mini.getContext("2d");

const TILE = 30;
const WORLD_SIZE = 40;
const VIEW = 20;

let gameRunning = false;

let player = { x: 10, y: 10, hp: 10, atk: 1 };
let enemies = [];
let boxes = [];
let particles = [];

function rand(max){ return Math.floor(Math.random()*max); }

function spawn(){
  enemies = Array.from({length:10},()=>({x:rand(WORLD_SIZE),y:rand(WORLD_SIZE)}));
  boxes = Array.from({length:15},()=>({x:rand(WORLD_SIZE),y:rand(WORLD_SIZE)}));
}

// 💾 SAVE / LOAD
function save(){
  localStorage.setItem("larp_save", JSON.stringify({player,enemies,boxes}));
}

function load(){
  const data = localStorage.getItem("larp_save");
  if(data){
    const s = JSON.parse(data);
    player = s.player;
    enemies = s.enemies;
    boxes = s.boxes;
  }
}

function startGame(newGame){
  document.getElementById("menu").style.display="none";
  document.getElementById("death").style.display="none";
  canvas.style.display="block";
  mini.style.display="block";
  document.getElementById("ui").style.display="block";

  gameRunning = true;

  if(newGame){
    player = { x:10,y:10,hp:10,atk:1 };
    spawn();
  } else {
    load();
    if(!enemies.length) spawn();
  }

  loop();
}

function goMenu(){
  gameRunning = false;
  canvas.style.display="none";
  mini.style.display="none";
  document.getElementById("ui").style.display="none";
  document.getElementById("menu").style.display="block";
  document.getElementById("death").style.display="none";
}

function restart(){
  startGame(true);
}

function move(dx,dy){
  if(!gameRunning) return;

  player.x = Math.max(0,Math.min(WORLD_SIZE-1,player.x+dx));
  player.y = Math.max(0,Math.min(WORLD_SIZE-1,player.y+dy));
  checkBoxes();
}

function attack(){
  if(!gameRunning) return;

  enemies = enemies.filter(e=>{
    const hit = Math.abs(e.x-player.x)<=1 && Math.abs(e.y-player.y)<=1;
    if(hit) particles.push({x:e.x,y:e.y,t:10});
    return !hit;
  });
}

function checkBoxes(){
  boxes = boxes.filter(b=>{
    if(b.x===player.x && b.y===player.y){ player.hp++; return false; }
    return true;
  });
}

// 🐌 SLOWER ENEMY AI (FIXED)
function updateEnemies(){
  enemies.forEach(e=>{
    // EXTREMELY SLOW ENEMY AI
    if(Math.random()<0.05){
      if(player.x>e.x) e.x++;
      else if(player.x<e.x) e.x--;

      if(player.y>e.y) e.y++;
      else if(player.y<e.y) e.y--;
    }
  });

  enemies.forEach(e=>{
    if(e.x===player.x && e.y===player.y){
      player.hp -= 1;
    }
  });
}

function updateParticles(){
  particles.forEach(p=>p.t--);
  particles = particles.filter(p=>p.t>0);
}

function cam(){
  return {
    x: player.x - Math.floor(VIEW/2),
    y: player.y - Math.floor(VIEW/2)
  };
}

function drawMinimap(){
  mctx.clearRect(0,0,150,150);
  const scale = 150 / WORLD_SIZE;

  mctx.fillStyle="cyan";
  mctx.fillRect(player.x*scale,player.y*scale,3,3);

  mctx.fillStyle="red";
  enemies.forEach(e=>mctx.fillRect(e.x*scale,e.y*scale,2,2));

  mctx.fillStyle="yellow";
  boxes.forEach(b=>mctx.fillRect(b.x*scale,b.y*scale,2,2));
}

function draw(){
  ctx.clearRect(0,0,600,600);

  const c = cam();

  for(let y=0;y<VIEW;y++){
    for(let x=0;x<VIEW;x++){
      const wx = x + c.x;
      const wy = y + c.y;

      const px = x*TILE;
      const py = y*TILE;

      ctx.strokeStyle="#222";
      ctx.strokeRect(px,py,TILE,TILE);

      if(wx===player.x && wy===player.y) ctx.fillText("🙂",px+10,py+20);
      if(enemies.some(e=>e.x===wx&&e.y===wy)) ctx.fillText("👾",px+10,py+20);
      if(boxes.some(b=>b.x===wx&&b.y===wy)) ctx.fillText("📦",px+10,py+20);

      particles.forEach(p=>{
        if(p.x===wx&&p.y===wy) ctx.fillText("💥",px+10,py+20);
      });
    }
  }

  ctx.fillStyle="white";
  ctx.fillText("HP: " + player.hp + " | ENEMIES: " + enemies.length,10,15);

  drawMinimap();
}

function checkDeath(){
  if(player.hp <= 0 && gameRunning){
    gameRunning = false;
    document.getElementById("death").style.display="block";
  }
}

function loop(){
  if(!gameRunning) return;

  updateEnemies();
  updateParticles();
  draw();
  save();
  checkDeath();

  requestAnimationFrame(loop);
}

// CONTROLS
document.addEventListener("keydown",e=>{
  if(e.key==='w')move(0,-1);
  if(e.key==='s')move(0,1);
  if(e.key==='a')move(-1,0);
  if(e.key==='d')move(1,0);
  if(e.key===' ')attack();
});
</script>
